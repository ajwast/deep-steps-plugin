cmake_minimum_required(VERSION 3.22)

# System Configuration
if(APPLE)
    set(CMAKE_OSX_DEPLOYMENT_TARGET "12.7" CACHE STRING "Minimum macOS version")
endif()

# Project Setup
project(DeepSteps VERSION 0.0.1)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# JUCE Setup 
set(JUCE_DIR "/Users/path/to/JUCE")
add_subdirectory(${JUCE_DIR} "${CMAKE_BINARY_DIR}/JUCE")

# Create the Plugin Target
juce_add_plugin(DeepSteps
    COMPANY_NAME "AW"
    IS_SYNTH TRUE
    NEEDS_MIDI_INPUT TRUE
    NEEDS_MIDI_OUTPUT TRUE
    IS_MIDI_EFFECT FALSE
    COPY_PLUGIN_AFTER_BUILD TRUE
    PLUGIN_MANUFACTURER_CODE AWDS
    PLUGIN_CODE Ds01
    FORMATS AU VST3 Standalone
    PRODUCT_NAME "Deep Steps")

# Organise Files
set(SourceFiles
        src/PluginProcessor.cpp
        src/PluginProcessor.h
        src/PluginEditor.cpp
        src/PluginEditor.h
        src/CustomLookAndFeel.cpp
        src/CustomLookAndFeel.h
        src/Autoencoder.h
        src/GrooveNN.h
        src/RNN.h
)

source_group("Source" FILES ${SourceFiles})

# Add Sources to Target
target_sources(DeepSteps
    PRIVATE
        ${SourceFiles})

# LibTorch Setup
set(CMAKE_PREFIX_PATH "path/to/libtorch")
find_package(Torch REQUIRED)

# Compile Definitions
target_compile_definitions(DeepSteps
    PUBLIC
        JUCE_WEB_BROWSER=0
        JUCE_USE_CURL=0
        JUCE_VST3_CAN_REPLACE_VST2=0)

# Add fonts as Binary data
juce_add_binary_data(DeepStepsAssets
        HEADER_NAME "BinaryData.h"
        SOURCES
        assets/fonts/LeagueSpartanRegular.ttf
        assets/fonts/LeagueSpartanBold.ttf
)

# Linking
target_link_libraries(DeepSteps
    PRIVATE
        juce::juce_audio_utils
        juce::juce_dsp
        juce::juce_audio_formats
        ${TORCH_LIBRARIES}
        DeepStepsAssets
    PUBLIC
        juce::juce_recommended_config_flags
        juce::juce_recommended_lto_flags
        juce::juce_recommended_warning_flags)

juce_generate_juce_header(DeepSteps)
